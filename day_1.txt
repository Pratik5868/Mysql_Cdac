delete from dept

commit;

--truncate the table
all the records will be delete but the structure is preserved 
can not give where condition
can not rollback operation

truncate table dept;

--diff in delete and truncate
delete :  will delete the record rowwise
truncate : will delete all rows

delete : rollback is possible
truncate : rollback is not possible

delete : can give "where" condition
truncate : can not give "where" condition

delete : slow
truncate : faster 

delete : dml
truncate : ddl

delete : trigger can be used with "delete" operation
truncate : trigger can not be used with "truncate" operation

---to drop table
to delete the structure and the records 
drop table dept;

--diff in drop and truncate
truncate : to delete the records but the structure is preserved
drop : to delete the structure and the records both together
